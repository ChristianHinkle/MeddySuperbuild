cmake_minimum_required(VERSION 4.0)

project(MeddySDKSuperbuildStandalone
  LANGUAGES CXX
  )

include(ExternalProject)

set(CMAKE_ARGS_FOR_SUB_BUILDS)

if(CMAKE_BUILD_TYPE)
  list(APPEND CMAKE_ARGS_FOR_SUB_BUILDS -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE})
endif()

if(CMAKE_INSTALL_PREFIX)
  list(APPEND CMAKE_ARGS_FOR_SUB_BUILDS -DCMAKE_INSTALL_PREFIX=${CMAKE_INSTALL_PREFIX})
endif()

if(CMAKE_PREFIX_PATH)
  list(APPEND CMAKE_ARGS_FOR_SUB_BUILDS -DCMAKE_PREFIX_PATH=${CMAKE_PREFIX_PATH})
endif()

ExternalProject_Add(CppUtils
  SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/CppUtils"
  CMAKE_ARGS ${CMAKE_ARGS_FOR_SUB_BUILDS}
  )

ExternalProject_Add(Boost
  SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/boost"
  CMAKE_ARGS ${CMAKE_ARGS_FOR_SUB_BUILDS}
  )

ExternalProject_Add(MeddySDK
  DEPENDS CppUtils
  DEPENDS Boost
  SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/MeddySDK"
  CMAKE_ARGS ${CMAKE_ARGS_FOR_SUB_BUILDS}
  )

ExternalProject_Add(MeddyCLI
  DEPENDS MeddySDK
  SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/MeddyCLI"
  CMAKE_ARGS ${CMAKE_ARGS_FOR_SUB_BUILDS}
  )

# Unset this variable, as it is no longer needed.
unset(CMAKE_ARGS_FOR_SUB_BUILDS)
